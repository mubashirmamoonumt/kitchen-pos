git clone https://github.com/mubashirmamoonumt/kitchen-pos.git
cd kitchen-pos

# 3. Create .env file for the backend

DATABASE_URL=postgresql://postgres:Mufaz2024kitchen@db.cqowbsiivxnckpkupedn.supabase.co:5432/postgres?sslmode=require
JWT_SECRET=MufazKitchen_SuperSecret_2024_POS!
NODE_ENV=production
PORT=3000
EOF

# 4. Install dependencies
pnpm install

# 5. Push schema to Supabase (creates all tables)
DATABASE_URL=postgresql://postgres:Mufaz2024kitchen@db.cqowbsiivxnckpkupedn.supabase.co:5432/postgres?sslmode=require \
  pnpm --filter @workspace/db run push-force

# 6. Start the backend (with PM2 so it stays running)
npm install -g pm2
cd artifacts/api-server
pm2 start "pnpm run dev" --name mufaz-api
pm2 save
pm2 startup
